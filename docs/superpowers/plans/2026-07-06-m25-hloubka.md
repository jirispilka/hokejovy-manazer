```ts
import type { GameState, Hrac, Liga } from './types'

export function utocneLadeni(h: Hrac): number {
  const a = h.atributy
  return Math.round(a.strelba * 0.4 + a.technika * 0.3 + a.prihravky * 0.3)
}

export function obranneLadeni(h: Hrac): number {
  const a = h.atributy
  return Math.round(a.obrana * 0.5 + a.fyzicka * 0.3 + a.brusleni * 0.2)
}

export type Role = 'Střelec' | 'Tvůrce hry' | 'Dvoucestný' | 'Univerzál'

export function roleHrace(h: Hrac): Role | null {
  if (h.pozice === 'G') return null
  if (h.atributy.strelba >= h.atributy.prihravky + 10) return 'Střelec'
  if (h.atributy.prihravky >= h.atributy.strelba + 10) return 'Tvůrce hry'
  if (obranneLadeni(h) >= utocneLadeni(h) - 5) return 'Dvoucestný'
  return 'Univerzál'
}

export type VysledekZnak = 'V' | 'VP' | 'PP' | 'P'

export function formaTymu(liga: Liga, klubId: string): VysledekZnak[] {
  return liga.zapasy
    .filter((z) => z.vysledek && (z.domaci === klubId || z.hoste === klubId))
    .sort((a, b) => a.den - b.den)
    .slice(-5)
    .map((z) => {
      const v = z.vysledek!
      const domaci = z.domaci === klubId
      const vyhra = domaci ? v.golyDomaci > v.golyHoste : v.golyHoste > v.golyDomaci
      const poProdlouzeni = v.prodlouzeni || v.najezdy
      return vyhra ? (poProdlouzeni ? 'VP' : 'V') : poProdlouzeni ? 'PP' : 'P'
    })
}

export interface RadekBodovani {
  jmeno: string
  klubId: string
  goly: number
  asistence: number
  body: number
}

export function kanadskeBodovani(s: GameState, liga: Liga): RadekBodovani[] {
  const radky: RadekBodovani[] = []
  for (const klubId of liga.tymy) {
    for (const h of s.tymy[klubId].hraci) {
      if (h.goly + h.asistence === 0) continue
      radky.push({
        jmeno: `${h.jmeno} ${h.prijmeni}`,
        klubId,
        goly: h.goly,
        asistence: h.asistence,
        body: h.goly + h.asistence,
      })
    }
  }
  return radky.sort((a, b) => b.body - a.body || b.goly - a.goly).slice(0, 10)
}
```

```ts
export function osobniProslov(
  stav: StavZapasu,
  strana: 'domaci' | 'hoste',
  tym: Tym,
  hracId: string,
  volba: 'povzbudit' | 'zdrbat',
  rng: Rng,
): StavZapasu {

    throw new Error('Osobní proslov jde jen v pauze mezi třetinami.')
  if (stav[strana].osobniProslovPouzit.includes(hracId))

  const hrac = tym.hraci.find((h) => h.id === hracId)

  const st = s[strana]
  st.osobniProslovPouzit.push(hracId)
  let bonus = 1
  let energieDelta = 0
  let text = `${hrac.jmeno} ${hrac.prijmeni} bere slova trenéra na vědomí.`

      energieDelta = 8

    }
  } else {

    if (r < 0.4) {

      energieDelta = 10
      text = `🔥 ${hrac.jmeno} ${hrac.prijmeni} dostal kartáč — a jde si to vyříkat s puky!`
    } else if (r >= 0.75) {
      bonus = 0.94
      energieDelta = -5
      text = `😬 ${hrac.jmeno} ${hrac.prijmeni} kritiku nese špatně.`
    }
  }
  st.osobniBonus[hracId] = bonus
  st.energie[hracId] = clamp((st.energie[hracId] ?? 100) + energieDelta, 0, 100)
  s.udalosti.push({ minuta: s.minuta, typ: 'proslov', tymId: st.klubId, hracId, text })
  return s
}

export function upravSestavuVZapase(
  stav: StavZapasu,
  strana: 'domaci' | 'hoste',
  tym: Tym,
  novaSestava: Sestava,
): StavZapasu {
  if (stav.faze !== 'pauza1' && stav.faze !== 'pauza2')
    throw new Error('Lajny jdou přeskládat jen v pauze mezi třetinami.')
  const st = stav[strana]
  const podleId = new Map(tym.hraci.map((h) => [h.id, h]))
  const vsichni = [...novaSestava.utoky.flat(), ...novaSestava.obrany.flat(), novaSestava.brankar]
  if (novaSestava.utoky.length !== 4 || novaSestava.utoky.some((l) => l.length !== 3))
    throw new Error('Sestava musí mít 4 útoky po 3 hráčích.')
  if (novaSestava.obrany.length !== 3 || novaSestava.obrany.some((l) => l.length !== 2))
    throw new Error('Sestava musí mít 3 obrany po 2 hráčích.')
  if (new Set(vsichni).size !== vsichni.length) throw new Error('Hráč nesmí být v sestavě dvakrát.')
  for (const id of vsichni) {
    const h = podleId.get(id)
    if (!h) throw new Error('Hráč není v týmu.')
    if (h.zranenZapasu > 0 || st.zraneni.includes(id)) throw new Error(`${h.jmeno} ${h.prijmeni} je zraněný.`)
  }

    if (podleId.get(id)!.pozice !== 'U') throw new Error('V útoku smí hrát jen útočníci.')

  for (const l of novaSestava.obrany) for (const id of l) {
    if (podleId.get(id)!.pozice !== 'D') throw new Error('V obraně smí hrát jen obránci.')
  }
  if (podleId.get(novaSestava.brankar)!.pozice !== 'G') throw new Error('V bráně musí být brankář.')

  const s = structuredClone(stav)
  const cil = s[strana]
  const stary = otiskLajn(cil.sestava)
  const novy = otiskLajn(novaSestava)
  cil.chemie = {
    utoky: novy.utoky.map((o, i) => (o === stary.utoky[i] ? cil.chemie.utoky[i] : 30)),
    obrany: novy.obrany.map((o, i) => (o === stary.obrany[i] ? cil.chemie.obrany[i] : 30)),
  }
  cil.sestava = structuredClone(novaSestava)
  s.udalosti.push({ minuta: s.minuta, typ: 'info', tymId: cil.klubId, text: 'Trenér přeskládal lajny.' })
  return s
}
```

```tsx
  const vPauze = stav.faze === 'pauza1' || stav.faze === 'pauza2'
  const podleId = new Map(tym.hraci.map((h) => [h.id, h]))
  const vSestave = new Set([...strana.sestava.utoky.flat(), ...strana.sestava.obrany.flat(), strana.sestava.brankar])

  function poklik(id: string) {
    if (!vPauze) return
    const hrac = podleId.get(id)!
    if (hrac.zranenZapasu > 0 || strana.zraneni.includes(id)) return
    if (!vybrany) return setVybrany(id)
    if (vybrany === id) return setVybrany(null)
    if (podleId.get(vybrany)!.pozice !== hrac.pozice) return setVybrany(id)
    setStav(upravSestavuVZapase(stav, mojeStrana, tym, vymenVSestave(strana.sestava, vybrany, id)))
    setVybrany(null)
  }

  const hodnoceniBarva = (h: number) => (h >= 7.5 ? 'var(--vyhra)' : h >= 5.5 ? 'var(--text)' : 'var(--prohra)')

  const Radek = ({ id }: { id: string }) => {
    const h = podleId.get(id)!
    const zranenyVZapase = strana.zraneni.includes(id) || h.zranenZapasu > 0
    const proslovByl = strana.osobniProslovPouzit.includes(id)
    return (
      <div
        className={`radek-hrace ${vPauze && !zranenyVZapase ? 'klik' : ''} ${vybrany === id ? 'vybrany' : ''} ${zranenyVZapase ? 'zraneny' : ''}`}
        onClick={() => poklik(id)}
        style={{ gridTemplateColumns: '26px minmax(140px, 1fr) 40px 70px 40px 70px 84px' }}
      >
        <BadgePozice pozice={h.pozice} />
        <span>
          {h.jmeno} {h.prijmeni}
          {zranenyVZapase && ' 🚑'}
        </span>
        <span className="radek-cislo">{h.vek} l.</span>
        <MiniBar hodnota={strana.energie[id] ?? 100} popisek="E" barva="var(--zlata)" />
        <b style={{ color: hodnoceniBarva(strana.hodnoceni[id] ?? 6) }}>
          {(strana.hodnoceni[id] ?? 6).toFixed(1)}
        </b>
        <MiniBar hodnota={h.forma} popisek="F" />
        <span>
          {vPauze && !zranenyVZapase && (
            <>
              <button
                className="tlacitko sekundarni tlacitko-mini"
                disabled={proslovByl}
                title="Povzbudit"
                onClick={(e) => {
                  e.stopPropagation()
                  setStav(osobniProslov(stav, mojeStrana, tym, id, 'povzbudit', rng()))
                }}
              >
                💪
              </button>{' '}
              <button
                className="tlacitko sekundarni tlacitko-mini"
                disabled={proslovByl}
                title="Zdrbat"
                onClick={(e) => {
                  e.stopPropagation()
                  setStav(osobniProslov(stav, mojeStrana, tym, id, 'zdrbat', rng()))
                }}
              >
                😠
              </button>
            </>
          )}
        </span>
      </div>
    )
  }

  return (
    <div className="karta">
      <h3>Moji hráči {vPauze && <span className="pill">pauza: klikni na dva hráče stejné pozice = výměna</span>}</h3>
      {strana.sestava.utoky.map((lajna, i) => (
        <div key={i}>
          <div style={{ color: 'var(--tlumeny)', fontSize: 12, margin: '6px 0 2px' }}>{i + 1}. útok</div>
          {lajna.map((id) => (
            <Radek key={id} id={id} />
          ))}
        </div>
      ))}
      {strana.sestava.obrany.map((dvojice, i) => (
        <div key={i}>
          <div style={{ color: 'var(--tlumeny)', fontSize: 12, margin: '6px 0 2px' }}>{i + 1}. obrana</div>
          {dvojice.map((id) => (
            <Radek key={id} id={id} />
          ))}
        </div>
      ))}
      <div style={{ color: 'var(--tlumeny)', fontSize: 12, margin: '6px 0 2px' }}>Brankář</div>
      <Radek id={strana.sestava.brankar} />
      <div style={{ color: 'var(--tlumeny)', fontSize: 12, margin: '6px 0 2px' }}>Náhradníci</div>
      {tym.hraci
        .filter((h) => !vSestave.has(h.id))
        .map((h) => (
          <Radek key={h.id} id={h.id} />
        ))}
    </div>
  )
}

function HvezdaZapasu({ stav, domaci, hoste }: { stav: StavZapasu; domaci: Tym; hoste: Tym }) {
  const vsichni = [...domaci.hraci, ...hoste.hraci]
  let nejlepsi: { jmeno: string; hodnoceni: number } | null = null
  for (const h of vsichni) {
    const hod = stav.domaci.hodnoceni[h.id] ?? stav.hoste.hodnoceni[h.id]
    if (hod !== undefined && (!nejlepsi || hod > nejlepsi.hodnoceni)) {
      nejlepsi = { jmeno: `${h.jmeno} ${h.prijmeni}`, hodnoceni: hod }
    }
  }
  if (!nejlepsi) return null
  return (
    <p>
      ⭐ Hvězda zápasu: <b>{nejlepsi.jmeno}</b> ({nejlepsi.hodnoceni.toFixed(1)})
    </p>
  )
}
```
